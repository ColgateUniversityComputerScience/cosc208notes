# Example program
```C
int main(int argc, char **argv) {
    printf("Begin\n");
    int pid = fork();
    if (pid == 0) {
        printf("Child\n");
        char *cmd[] = { "/usr/bin/date", NULL };
        execv(cmd[0], cmd);
    } else {
        printf("Parent\n");
    }
    printf("End\n");
    return 0;
}
```

int main() {

    if (a == 0) {
        execv("./progB", NULL);
        printf("A 2nd gen\n");

        wait(NULL);
        printf("A 1st gen\n");
        return 0;
    }
}
