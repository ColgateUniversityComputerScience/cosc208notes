* Q3: _What does the following code output (assuming the new process has PID 1819)?_


```c
#include <stdio.h>
#include <unistd.h>    
int main(int argc, char **argv) {
    printf("Before fork\n");
    int pid = fork();
    if (pid < 0) {
        printf("Fork failed\n");
        return 1;
    } else if (pid == 0) {
        printf("Child gets %d\n", pid);
    } else {
        printf("Parent gets %d\n", pid);
    }
    return 0;
}
```

## Waiting for processes

* Q4: _What are all possible outputs of this program?_


```c
#include <stdio.h>
#include <unistd.h>
int main() {
    int pid = fork();
    if (pid < 0) {
        printf("Fork failed\n");
        return 1;
    } else if (pid == 0) {
        printf("Child\n");
    } else {
        wait(NULL);
        printf("Parent\n");
    }
    return 0;
}
```

## Extra practice

* Q6: _What does the following code output?_


```c
#include <stdio.h>
#include <unistd.h>
int main(int argc, char **argv) {
    int value = 100;
    int pid = fork();
    if (pid < 0) {
        printf("Fork failed\n");
        return 1;
    } else if (pid == 0) {
        value -= 50;
    } else {
        value += 50;
    }
    printf("My value is %d\n", value);
    return 0;
}
```
