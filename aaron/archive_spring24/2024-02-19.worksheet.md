## Pointer arithmetic

* Example


```c
#include <stdio.h>
int main() {
    int nums[] = {2,4,8,16,32,64,128};
    
    int *ptr = nums + 2;
    printf("%d %d\n", *ptr, *nums);
    ptr = ptr + 2;
    printf("%d %d\n", *ptr, *nums);
    // nums = nums + 2; // Not allowed; array variable must point to 0th element
    
    int val = *(nums + 2);
    printf("%d %d\n", val, *nums);
    val = *(ptr + 2);
    printf("%d %d\n", val, *ptr);
    
    nums[2] = 0;
    printf("%d %d\n", *nums, nums[2]);
    *(nums + 2) = -1;
    printf("%d %d\n", *nums, nums[2]);
}
```
