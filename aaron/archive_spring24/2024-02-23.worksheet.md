## Pointers as return values


```c
#include <stdio.h>
int *one() {
    int x = 1;
    int *p = &x;
    return p;
}
int main() {
    int *q = one();
    printf("%d\n", *q);
}
```

* Q8: _What is the output of this program?_


```c
#include <stdio.h>
#include <string.h>

void modify(char *str, int idx) {
    char tmp = str[0];
    idx--;
    str[0] = str[idx];
    str[idx] = tmp;
}

int main() {
    char a[5] = "none";
    char b[3] = "no";
    char *x = &(a[1]);
    printf("%s %s\n", b, x);

    char *y = b;
    int i = strlen(y);
    printf("%s %d\n", y, i);

    modify(y, i);
    printf("%s-%s-%s %d\n", x, y, x, i);

    *(a+2) = '\0';
    i = strlen(a);
    printf("%s %d\n", a, i);
}
```
