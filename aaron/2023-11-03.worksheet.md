    int x = 13;

* Q7: _What does the following code output (assuming the new process has PID 1819)?_


```c
#include <stdio.h>
#include <unistd.h>    
int main(int argc, char **argv) {
    printf("Before fork\n");
    int pid = fork();
    if (pid < 0) {
        printf("Fork failed\n");
        return 1;
    } else if (pid == 0) {
        printf("Child gets %d\n", pid);
    } else {
        printf("Parent gets %d\n", pid);
    }
    return 0;
}
```

    int value = 100;

        return 1;
