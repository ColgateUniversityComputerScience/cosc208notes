    int result = fork();
    if (result < 0) {
        return 1;
    } else if (result == 0) {
        printf("Child\n");
        char msg[1024];
        read(pipe_descriptors[0], msg, 1024);
        printf("%s\n", msg);
    } else {
        char msg[1024] = "Hello from your parent!";

    }

}
